#include <iostream>
#include <string>

using namespace std;

class stack {
private:
    int t, maxSize;
    char *arr;
public:
    stack(int size) {
        t = -1;
        maxSize = size;
        arr = new char[maxSize];
    }

    ~stack() {
        delete[] arr;
    }

    bool empty() {
        return (t == -1);
    }

    char top() {
        return arr[t];
    }

    char pop() {
        return arr[t--];
    }

    void push(char data) {
        arr[++t] = data;
    }
};

int main() {
    string n;
    cin >> n;
    int size = n.length();
    stack st(size);
    for (int i = 0; i < size; i++) {
        if ('0' < n[i] && n[i] <= '9') {
            cout << n[i];
        } else {
            if (n[i] == '*' || n[i] == '/') {
                while (!st.empty() && (st.top() == '*' || st.top() == '/')) {
                    cout << st.pop();
                }
                st.push(n[i]);
            } else if (n[i] == '+' || n[i] == '-') {
                while (!st.empty()) {
                    cout << st.pop();
                }
                st.push(n[i]);

            }
        }
    }

    while (!st.empty())
        cout << st.pop();
}
